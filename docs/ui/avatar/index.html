# Avatar 头像 (opens new window)

用来展示用户头像，及头像各种状态

# 按需引入


import Avatar from '@yy/slack-ui/dist/lib/avatar'
import '@yy/slack-ui/dist/lib/avatar/index.css'

# 基础用法

<template>
  <Avatar
    :class="$style.avatar"
    src="https://o-id.ihago.net/blob/v1/ali/id/uurl/1617875956701102.jpeg"
  ></Avatar>
</template>

<script lang="ts">
import { Vue, Component } from 'vue-property-decorator'
import Avatar from '@yy/slack-ui/dist/lib/avatar'
import '@yy/slack-ui/dist/lib/avatar/index.css'
@Component({
  components: {
    Avatar
  }
})
export default class test extends Vue {}
</script>
<style lang="scss" module>
.avatar {
  width: 100px;
  height: 100px;
}
</style>

# 自定义 icon 用法

<template>
  <Avatar
    :class="$style.avatar"
    src="https://o-id.ihago.net/blob/v1/ali/id/uurl/1617875956701102.jpeg"
  >
    <template slot="icon">
      <span :class="$style.icon"></span>
    </template>
  </Avatar>
</template>

<script lang="ts">
import { Vue, Component } from 'vue-property-decorator'
import Avatar from '@yy/slack-ui/dist/lib/avatar'
import '@yy/slack-ui/dist/lib/avatar/index.css'
@Component({
  components: {
    Avatar
  }
})
export default class test extends Vue {}
</script>
<style lang="scss" module>
.avatar {
  width: 100px;
  height: 100px;
}
.icon {
  display: block;
  width: 10px;
  height: 10px;
  border-radius: 100%;
  background: #cc0000;
  position: absolute;
  right: 0px;
  bottom: 0px;
}
</style>

# 参数

# props

Name Description Type Required Default
shape 头像形状 round / square String false round
src 头像地址 String false -
icon 头像默认icon是否开启 Boolean false false
border 头像border Boolean false false

# slots

Name Description Default Slot Content
img 图片插槽 默认为 img 样式
icon icon 插槽 默认为 dot样式
