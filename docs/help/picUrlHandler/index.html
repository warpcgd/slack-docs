# 项目本地图片链接适配avif或webp

# 背景

  • 该方法目前可对项目内图片链接进行avif或者webp图片的适配，项目需为fec构建并且开启@yy/fec-plugin-hago-cdn插件的enableOssUpload选项。

# 转换原理

# webp转换

  • fec构建打包后，会将图片以及css资源上传至阿里oss存储服务，该存储服务支持图片格式转换，对传入的url的处理就是加上符合oss存储服务格式转换 (opens new window)的参数来完成webp格式转换的目的。

# avif转换

  • fec构建打包后，会对图片进行转换压缩至avif格式，并上传至阿里oss存储服务，对传入的url的处理，替换图片格式后缀为avif。

从 Vue 3.0 开始，过滤器已移除，且不再支持。 所以不建议用filter的方式使用该插件。

# 引入


import { handlerPicUrl } from '@yy/slack-help'

# get方式使用样例

<template>
  <div>
    <img :src="testOssPic" alt="">
  </div>
</template>

<script lang="ts">
import { Vue, Component } from 'vue-property-decorator'
// 引入依赖
import { handlerPicUrl } from '@yy/slack-help'

// 引入项目图片依赖
import testPic from  './img/test.png'

@Component()
export default class test extends Vue {
  get testOssPic() {
    // 转换oss img链接
   return handlerPicUrl(testPic)
  }
}
</script>

# method方式使用样例

<template>
  <div>
    <img :src="convertOssWebp(testPic)" alt="">
  </div>
</template>

<script lang="ts">
import { Vue, Component } from 'vue-property-decorator'
// 引入依赖
import { handlerPicUrl } from '@yy/slack-help'

// 引入项目图片依赖
import testPic from  './img/test.png'

@Component()
export default class test extends Vue {
  testPic = testPic

  convertOssWebp(url: string){
    return handlerPicUrl(url)
  }
}
</script>
