# 根据不同的类型，动态修改分享参数

有可能会遇到一种业务场景，需要 facebook 和 whatsapp 分享不同的图片和文案，此时，需要借助组件的 before-share 事件，来达成目的

<template>
  <div>
    <Share
      :share-types="[1, 2, 3, 10]"
      :share-opts="shareOpts"
      @before-share="onBeforeShare"
    ></Share>
  </div>
</template>

<script lang="ts">
import { Vue, Component } from 'vue-property-decorator'
import { Share } from '@yy/slack-business/dist/lib/share/index'
import '@yy/slack-business/dist/lib/share/index.css'

@Component({
  components: {
    Share
  }
})
export default class test extends Vue {
  shareOpts = {
    link: 'https://www.ihago.net/a/aflink-proxy/index.html',
    activityId: 'xx',
    title: 'demo-title',
    image: 'https://o-static.ihago.net/ikxd/b024bb3815a1bbd7f4fc7b91840c88e8/600314yingyu.png',
    content: 'demo-content'
  }

  /**
   * @param type 对应分享的 ShareTypes
   * @param resolve 回调方法，一旦监听了 before-share 这个事件，必须手动执行该方法才能完成整个分享
   */
  onBeforeShare(type, resolve) {
    if (type === 10) {
      // 传入参数是 ShareOpts 类型
      resolve({
        ...this.shareOpts,
        title: 'change-by-before-share'
      })
    }
    resolve()
  }
}
</script>

# 参数定义

// ShareTypes 分享类型定义
enum ShareTypes {
  LINE = 1,
  WHATSAPP = 2,
  INSTAGRAM = 3,
  FACEBOOK = 5,
  MESSENGER = 6,
  VK = 9,
  COPYLINK = 10,
  ZALO = 11,
  OTHERS = 0
}

// 分享参数
interface ShareOpts {
  link?: string // url链接（不带参数，非短链）
  fullLink?: string // 完整链接，非短链
  activityId: string // 活动ID
  image: string // 分享图
  title: string // 分享标题
  content?: string // 分享文案，该字段会传递给客户端，但不会拼接到 URL 中
  queryExtend?: Record<string, string | number> // 传入 k-v 会补充到 link 后作为参数
}

# event

事件名 说明 回调参数
share 点击分享触发（注：因为客户端也没法捕捉到是否分享成功，所以建议我们不要依赖他们的回调，但基本都是成功的，因此也可将此事件直接当成分享成功处理） opts(端内分享参数)
before-share 点击分享，在形成分享参数之前触发 type: 分享类型, resolve: 回调方法，必须调用，可以将更新的参数传入
