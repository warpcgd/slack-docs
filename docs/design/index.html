# 设计规范

# 介绍

@yy/slack-ui 是基于Hago 视觉规范 (opens new window)实现的活动组件库

# 字体

  1. 字体
font-family: system-ui, -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, Ubuntu, 'Helvetica
Neue', sans-serif;

  1. 字号 默认 24px

  2. 行高 默认 lineheight: 1

# 组件

# modal 弹窗

  1. 进入/进出动效
  • duration: 300ms
  • animation: linear
  • 全局遮罩: #000000 透明度 0%→70%
  • 进入弹窗: 尺寸 50% → 110% → 100% 透明度 0% → 100%
  • 进出弹窗: 尺寸 100% → 0% 透明度 100% → 0%

DEMO

<template>
  <div>
    <button @click="val = true">Modal DEMO</button>

</template>

import '@yy/slack-ui/dist/lib/modal/index.css'
@Component({
  components: {
    Modal
  }
})
export default class ModalTest extends Vue {
  val = false
}
</script>

# toast 提示规范

  • 背景：#000000 透明度 80% 圆角 5px 高度 40px
  • 文字：14px
  • 时长：2s
  • 动画：透明度 0% → 80% 位置 -100% → 0%

DEMO

<template>
  <div>
    <button @click="showText">文字提示</button>
    <button @click="showLongText">长文字提示</button>
    <button @click="showSuccess">成功提示</button>
    <button @click="showError">失败提示</button>
    <button @click="showWarn">警告提示</button>
  </div>
</template>
<script lang="ts">
import { Vue, Component } from 'vue-property-decorator'
import Toast from '@yy/slack-ui/dist/lib/toast'
import '@yy/slack-ui/dist/lib/toast/index.css'

export default class test1 extends Vue {
  showText() {
    Toast.text('文字提示')
  }

  showLongText() {
    Toast.text(
      'Slack 是 Hago 前端开发和维护的一套 vue 组件库，分为业务组件@yy/slack-business， ui 组件@yy/slack-ui 和常用工具@yy/slack-help，支撑 Hago 活动页面'
    )
  }

  showSuccess() {
    Toast.success('成功提示')
  }

  showError() {
    Toast.error('失败提示')
  }

  showWarn() {
    Toast.warn('警告提示')
  }
}
</script>
